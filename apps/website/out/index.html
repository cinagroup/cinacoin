example.ts
import { Cinacoin } from '@cinacoin/sdk'

// Initialize the client
const client = Cinacoin.init({
  projectId: 'your-id',
  chains: [1, 137, 42161],
})

// Connect a wallet
const session = await client.connect()

// Send a cross-chain transaction
await client.transfer({
  to: '0x...',
  chain: 137,
})
